#Fungsi kategori_nilai(nilai) yang mengembalikan huruf A/B/C/D.

def kategori_nilai(nilai):
    if nilai < 0 or nilai > 100:
        return "??"
    
    if nilai < 30:
        return "D"
    
    if nilai < 50:
        return "C"
    
    if nilai < 80:
        return "B"
    
    if nilai <= 100:
        return "A"

nilai = kategori_nilai(35)
print(nilai)
