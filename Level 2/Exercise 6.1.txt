sample1 = "Belajar Python itu Seru Sekali"
sample1 = "".join(sample1.split()).lower()
dict_test = {}


for char in sample1:
    try:
        dict_test[char] = dict_test[char] + 1
    except KeyError:
        dict_test[char] = 1
